# PostgreSQL Database Schema Documentation
# Project: School Management System (MLZS)
# Date: 2026-02-02

-- Note: This schema is derived from the frontend application requirements and mock data structures.
-- Ideally, use an ORM like Prisma or Drizzle to manage this schema in code.

-- ENUMS --
CREATE TYPE "UserRole" AS ENUM ('SUPER_ADMIN', 'ADMIN', 'PRINCIPAL', 'TEACHER', 'STUDENT', 'ACCOUNTANT', 'PARENT');
CREATE TYPE "Gender" AS ENUM ('MALE', 'FEMALE', 'OTHER');
CREATE TYPE "AcademicYearStatus" AS ENUM ('ACTIVE', 'UPCOMING', 'ENDED');
CREATE TYPE "FeeFrequency" AS ENUM ('MONTHLY', 'QUARTERLY', 'HALF_YEARLY', 'YEARLY', 'ONCE');
CREATE TYPE "PaymentMethod" AS ENUM ('ONLINE', 'CASH', 'CHEQUE', 'Theft');
CREATE TYPE "PaymentStatus" AS ENUM ('PENDING', 'COMPLETED', 'FAILED');
CREATE TYPE "AttendanceStatus" AS ENUM ('PRESENT', 'ABSENT', 'LATE', 'HALF_DAY');
CREATE TYPE "NoticeAudience" AS ENUM ('ALL', 'STUDENTS', 'TEACHERS', 'PARENTS', 'CLASSROOM');

-- TABLES --

-- 1. USERS & AUTH
-- Central table for all system users
CREATE TABLE "User" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "email" VARCHAR(255) UNIQUE NOT NULL,
    "password" VARCHAR(255) NOT NULL, -- Hashed
    "name" VARCHAR(255) NOT NULL,
    "role" "UserRole" NOT NULL DEFAULT 'STUDENT',
    "image" TEXT,
    "isFirstLogin" BOOLEAN DEFAULT TRUE,
    "createdAt" TIMESTAMP DEFAULT NOW(),
    "updatedAt" TIMESTAMP DEFAULT NOW()
);

-- 2. ACADEMIC MANAGEMENT
-- Manages school years (e.g., 2025-2026)
CREATE TABLE "AcademicYear" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "name" VARCHAR(50) NOT NULL, -- e.g. "2025-2026"
    "startDate" DATE NOT NULL,
    "endDate" DATE NOT NULL,
    "status" "AcademicYearStatus" DEFAULT 'UPCOMING',
    "isCurrent" BOOLEAN DEFAULT FALSE
);

-- Classes (e.g., Class 1, Class 10)
CREATE TABLE "Classroom" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "name" VARCHAR(50) NOT NULL,
    "level" VARCHAR(50), -- Primary, Secondary, etc.
    "capacity" INTEGER DEFAULT 40,
    "academicYearId" UUID REFERENCES "AcademicYear"("id"),
    "classTeacherId" UUID REFERENCES "TeacherProfile"("id") -- One teacher assigned as class teacher
);

-- Sections (e.g., Section A, B)
CREATE TABLE "Section" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "name" VARCHAR(10) NOT NULL, -- "A", "B"
    "classroomId" UUID REFERENCES "Classroom"("id") ON DELETE CASCADE,
    "capacity" INTEGER DEFAULT 40
);

-- Subjects (e.g., Math, Science)
CREATE TABLE "Subject" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "name" VARCHAR(100) NOT NULL,
    "code" VARCHAR(20),
    "classroomId" UUID REFERENCES "Classroom"("id") ON DELETE CASCADE
    -- Note: Teacher assignment to subject is often M:N or 1:N. See SubjectTeacher table if M:N needed, or "teacherId" here if simple.
);

-- 3. PROFILES

-- Teacher Profile (Extends User)
CREATE TABLE "TeacherProfile" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "userId" UUID UNIQUE REFERENCES "User"("id") ON DELETE CASCADE,
    "designation" VARCHAR(100),
    "qualification" TEXT[], -- Array of strings
    "joiningDate" DATE,
    "specialization" VARCHAR(255),
    "phoneNumber" VARCHAR(20)
);

-- Student Profile (Extends User)
CREATE TABLE "StudentProfile" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "userId" UUID UNIQUE REFERENCES "User"("id") ON DELETE CASCADE,
    "admissionNo" VARCHAR(50) UNIQUE NOT NULL,
    "rollNumber" INTEGER,
    
    -- Classroom & Section Assignment
    "classroomId" UUID REFERENCES "Classroom"("id"),
    "sectionId" UUID REFERENCES "Section"("id"),
    
    -- Personal Info
    "dob" DATE NOT NULL,
    "gender" "Gender" NOT NULL,
    "nationality" VARCHAR(50) DEFAULT 'Indian',
    "address" TEXT,
    "bloodGroup" VARCHAR(5),
    
    -- Parent Info
    "fatherName" VARCHAR(255),
    "motherName" VARCHAR(255),
    "parentContact" VARCHAR(20),
    "parentEmail" VARCHAR(255),
    
    "joiningDate" DATE DEFAULT NOW()
);

-- Subject Assignment (Which teacher teaches which subject)
CREATE TABLE "SubjectTeacher" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "subjectId" UUID REFERENCES "Subject"("id") ON DELETE CASCADE,
    "teacherId" UUID REFERENCES "TeacherProfile"("id") ON DELETE CASCADE,
    "sectionId" UUID REFERENCES "Section"("id") -- Optional if subject is per class, but often teachers differ per section
);

-- 4. ACADEMICS & EXAMS

CREATE TABLE "Exam" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "name" VARCHAR(100) NOT NULL, -- "Mid-Term", "Finals"
    "date" TIMESTAMP NOT NULL,
    "durationMinutes" INTEGER,
    "totalMarks" INTEGER NOT NULL DEFAULT 100,
    "passingMarks" INTEGER DEFAULT 33,
    "subjectId" UUID REFERENCES "Subject"("id"),
    "academicYearId" UUID REFERENCES "AcademicYear"("id"),
    "isPublished" BOOLEAN DEFAULT FALSE
);

CREATE TABLE "ExamResult" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "examId" UUID REFERENCES "Exam"("id") ON DELETE CASCADE,
    "studentId" UUID REFERENCES "StudentProfile"("id") ON DELETE CASCADE,
    "marksObtained" DECIMAL(5, 2) NOT NULL,
    "grade" VARCHAR(5),
    "remarks" TEXT,
    "createdAt" TIMESTAMP DEFAULT NOW(),
    UNIQUE("examId", "studentId")
);

CREATE TABLE "Attendance" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "date" DATE NOT NULL,
    "studentId" UUID REFERENCES "StudentProfile"("id") ON DELETE CASCADE,
    "status" "AttendanceStatus" NOT NULL,
    "markedById" UUID REFERENCES "User"("id"), -- Teacher who marked it
    "remarks" TEXT,
    UNIQUE("date", "studentId")
);

-- 5. FEES & FINANCE

-- Fee definitions for a class/year
CREATE TABLE "FeeStructure" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "name" VARCHAR(100) NOT NULL, -- "Tuition Fee Q1"
    "amount" DECIMAL(10, 2) NOT NULL,
    "description" TEXT,
    "dueDate" DATE NOT NULL,
    "frequency" "FeeFrequency" DEFAULT 'MONTHLY',
    "classroomId" UUID REFERENCES "Classroom"("id"), -- If null, applies to all? Better to be explicit.
    "academicYearId" UUID REFERENCES "AcademicYear"("id")
);

-- Actual fee record for a student (Generated from FeeStructure)
CREATE TABLE "StudentFee" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "studentId" UUID REFERENCES "StudentProfile"("id") ON DELETE CASCADE,
    "feeStructureId" UUID REFERENCES "FeeStructure"("id"),
    "amountDue" DECIMAL(10, 2) NOT NULL,
    "amountPaid" DECIMAL(10, 2) DEFAULT 0,
    "status" VARCHAR(20) DEFAULT 'PENDING', -- PENDING, PAID, PARTIAL, OVERDUE
    "dueDate" DATE,
    "updatedAt" TIMESTAMP DEFAULT NOW()
);

CREATE TABLE "Payment" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "studentFeeId" UUID REFERENCES "StudentFee"("id"),
    "amount" DECIMAL(10, 2) NOT NULL,
    "date" TIMESTAMP DEFAULT NOW(),
    "method" "PaymentMethod" DEFAULT 'ONLINE',
    "transactionId" VARCHAR(100), -- For stripe/razorpay ref
    "status" "PaymentStatus" DEFAULT 'COMPLETED',
    "receiptUrl" TEXT
);

-- 6. COMMUNICATION

CREATE TABLE "Notice" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "title" VARCHAR(255) NOT NULL,
    "content" TEXT NOT NULL,
    "createdAt" TIMESTAMP DEFAULT NOW(),
    "authorId" UUID REFERENCES "User"("id"),
    "audience" "NoticeAudience" DEFAULT 'ALL',
    "targetClassroomId" UUID REFERENCES "Classroom"("id") -- Optional, for specific class notices
);

-- 7. ACCOUNTS (For Accountant Portal - Simplified)
-- Based on inference from directories seen (income/expenses)

CREATE TABLE "Expense" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "title" VARCHAR(255) NOT NULL,
    "amount" DECIMAL(10, 2) NOT NULL,
    "category" VARCHAR(50),
    "date" DATE NOT NULL,
    "description" TEXT,
    "approvedBy" UUID REFERENCES "User"("id")
);

CREATE TABLE "Income" (
    "id" UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    "source" VARCHAR(255) NOT NULL,
    "amount" DECIMAL(10, 2) NOT NULL,
    "date" DATE NOT NULL,
    "description" TEXT
);

-- INDEXES & CONSTRAINTS RECOMMENDATIONS
-- 1. Index on User(email)
-- 2. Index on StudentProfile(admissionNo)
-- 3. Composite Index on Attendance(studentId, date)
-- 4. Index on ExamResult(examId)
-- 5. Foreign Key Constraints are critical for data integrity.
